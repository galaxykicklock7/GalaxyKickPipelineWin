   WEB APP INTEGRATION:
================================================================================

const TUNNEL_URL = "https://xxxxx.trycloudflare.com";

// Configure
await fetch(`${TUNNEL_URL}/api/configure`, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    rc1: "Code1",
    rc2: "Code2",
    planet: "YourPlanet",
    device: "312",
    autorelease: true
  })
});

// Connect
await fetch(`${TUNNEL_URL}/api/connect`, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' }
});

// Get status
const status = await fetch(`${TUNNEL_URL}/api/status`).then(r => r.json());
console.log('Connected:', status.connected);
